# jpablobr .emacs.d #

## INSTALL

   % git clone git@github.com:jpablobr/emacs.d ~/.emacs.d

   % cd ~/.emacs.d

   % ruby install.rb

## IMPORTANT MODE TO COMPILE

   M-x nxhtmlmaint-start-byte-compilation

   M-x byte-compile js2-mode 

   M-x jde-compile-jde

   M-x ecb-byte-compile
